"use strict"
const $ = require('jquery')
const _ = require('lodash')
 
import {
    alphabetize,
    sendToBackground
} from './organize_glossaries'
 
export {removeDuplicates}
 
function removeDuplicateGlossaries(glossaryList) {
    let unique = []
    for (let i = 0; i < glossaryList.length; i++) {
        let currentGlossItem = glossaryList[i].innerHTML
        currentGlossItem = currentGlossItem.replace(/(gtc-gl-name-)[0-9]+/g, "$1number")
 
        if (unique.length > 0) {
            for (let j = 0, uniqueLength = unique.length; j < uniqueLength; j++) {
                if (unique[j] === currentGlossItem) {
                    glossaryList[i].parentNode.removeChild(glossaryList[i])
                    //i=0
                    break
                }
                if (j === uniqueLength - 1) {
                    unique.push(currentGlossItem)
                    break
                }
            }
        } else {
            unique.push(currentGlossItem)
        }
    }
}
 
function removeDuplicates() {
    /**
     *
     * @type {number}
     */
    const DEBOUNCE_DELAY = 25
 
    /**
     *
     * @type {*|jQuery|HTMLElement}
     */
    const $glossAreaArray = $('.gtc-tool-right-floating')
 
    /**
     *
     * @type {MutationObserver}
     */
    const obs1 = new MutationObserver(_.debounce(automated, DEBOUNCE_DELAY))
    /**
     *
     * @type {MutationObserver}
     */
 
    if (cth.option.removeDuplicates) {
        removeDuplicateGlossaries($glossAreaArray[0].querySelectorAll('.gtc-glossary-terms'))
        removeDuplicateGlossaries($glossAreaArray[1].querySelectorAll('.gtc-glossary-terms'))
    }
    const obs2 = new MutationObserver(_.debounce(custom, DEBOUNCE_DELAY))
    var automatedTranslatedSearch = $glossAreaArray[0]
    var customTranslationSearch = $glossAreaArray[1]
    var config = {childList: true, subtree: true}
    obs1.observe(automatedTranslatedSearch, config);
    obs2.observe(customTranslationSearch, config);
 
    function automated() {
        if ($glossAreaArray[0].length <= 1) return
        var that = this
        both(0, that)
    }
 
    function custom() {
        if ($glossAreaArray[1].length <= 1) return
        var that = this
        both(1, that)
    }
 
    function both(index, that) {
        that.disconnect()
        const glossArea = $('.gtc-tool-right-floating')[index]
        removeDuplicateGlossaries(glossArea.querySelectorAll('.gtc-glossary-terms'))
        alphabetize(glossArea.querySelectorAll('li.gtc-glossary-match-holder'))
        that.observe(glossArea, config)
        if (cth.option.GLOSSARY_WINDOW_TOGGLE === true) {
            sendToBackground()
        }
    }
}
