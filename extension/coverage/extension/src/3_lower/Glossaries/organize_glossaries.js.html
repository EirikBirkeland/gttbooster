"use strict"
const $ = require('jquery')
const _ = require('lodash')
 
import tinysort from 'tinysort'
const logger = logger || require('cth-logger')
 
function alphabetize() {
    let $glossItems = $('div.gtc-glossary-content-holder').find('li.gtc-glossary-match-holder')
 
    if ($glossItems && $glossItems.length && $glossItems.length > 1) {
        tinysort($glossItems)
        // _hideDefinition()
        highlightRelevant()
    }
 
    function _hideDefinition() {
        const $itemTohide = ".gtc-glossary-description"
        _.forEach($glossItems, ele => {
 
            const $elem = $(ele)
            const $desc = $elem.find($itemTohide)
 
            $desc.hide()
            $elem
                .mouseover(()=>$desc.show())
                .mouseleave(()=>$desc.hide())
        })
    }
 
    function highlightRelevant() {
        _.forEach($('.gtc-glossary-source'), ele => {
            if (ele.innerHTML.match(new RegExp("Source: " + cth.docInfo.produktNavn))) {
                ele.innerHTML = ele.innerHTML.replace(new RegExp("(" + cth.docInfo.produktNavn + ")"), `<span style="color:blue"><b>$1</b></span>`)
            }
        })
    }
}
 
function sendToBackground() {
    "use strict"
    let header = "glossaries"
    let body = _prepareGlossaryPage()
    let combined = {header, body}
    logger.debug(combined)
 
    chrome.runtime.sendMessage({header, body}, (response) => {
        logger.info(response)
    })
 
    function _prepareGlossaryPage() {
        const clonedDoc = document.cloneNode(true)
        const glossaryContentHolder = $(document).find('.gtc-tools-autosearch').find('.gtc-glossary-content-holder')
 
        try {
            var glossaryList = Array.from(glossaryContentHolder)[0].innerHTML
        } catch (err) {
            logger.info('gtc-glossary-content-holder was not ready?')
            window.cth.exceptionHandler(err.stack)
        }
 
        return glossaryList
    }
}
 
export {alphabetize, sendToBackground}
