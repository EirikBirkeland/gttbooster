 */
function escapeHtml(string) {
  const entityMap = Object.freeze({
    '&': '&amp;',
    '<': '&lt;',
    '>': '&gt;',
    '"': '&quot;',
    '\'': '&#39;',
    '/': '&#x2F;'
  })
 
  return String(string).replace(/[&<>"'\/]/g, s => {
    return entityMap[s]
  })
}
